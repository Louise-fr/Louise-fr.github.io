import pandas as pd

import matplotlib.pyplot as plt

iris_data_file_path = './iris.data'
iris_data = pd.read_csv(iris_data_file_path)
#iris.head()

X = iris_data[["sepal_length","sepal_width","petal_length","petal_width"]].values
y = iris_data["class"].values


# X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.20, random_state=42)

clf = svm.SVC()
# clf.fit(X_train, y_train)
scores = cross_val_score(clf, X, y, cv=5)
print(scores)
print("Accuracy: %0.2f (+/- %0.2f)" % (scores.mean(), scores.std() * 2))

#Wine SVM

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn import svm
from sklearn import datasets
from sklearn.linear_model import SGDClassifier 
from sklearn.model_selection import cross_val_score
from sklearn import preprocessing
import matplotlib.pyplot as plt

(data,target) = datasets.load_wine(return_X_y=True)

#X_train, X_test, y_train, y_test = train_test_split(data, target, test_size=0.20, random_state=42)

#SVM sans normalisation

clf = svm.SVC()
scores = cross_val_score(clf, data, target, cv=5)
#print(scores)
print("Accuracy: %0.2f (+/- %0.2f)" % (scores.mean(), scores.std() * 2))

#SVM avec normalisation

data_normalized = preprocessing.normalize(data, norm='l2')
data_scaled = preprocessing.scale(data_normalized)
# print(data)
# print(data_normalized)
# print(data_scaled)
clf = svm.SVC()
scores = cross_val_score(clf, data_scaled, target, cv=5, n_jobs=4)
print(scores)
print("Accuracy: %0.2f (+/- %0.2f)" % (scores.mean(), scores.std() * 2))
